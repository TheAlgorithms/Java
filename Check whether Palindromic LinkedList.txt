    { 
        Node one = new Node(1); 
        Node two = new Node(2);

        Node four = new Node(4);

        Node seven = new Node(1); 
        one.next = two;

        System.out.println("isPalidrome :" + condition);

    static boolean isPalindrome(Node head) 
    { 
  
        Node slow = head; 
        boolean ispalin = true; 
        Stack<Integer> stack = new Stack<Integer>(); 
  
        while (slow != null) { 
            stack.push(slow.data); 
            slow = slow.next; 
        } 
  
        while (head != null) { 
        int i = stack.pop(); 
            if (head.data == i) { 
                ispalin = true; 
            } 
            else { 
                ispalin = false; 
                break; 
            } 
            head = head.next; 
        } 
        return ispalin; 
    } 
} 
  
class Node { 
    int data; 
    Node next; 
    Node(int d) 
    { 
        next = null; 
        data = d; 
    } 
}
